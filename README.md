# VideoAll
    All video download


## Installation
    pip install videoall


## How to import
    from videoall import videoall


## Examples
    
    from videoall import videoall

    config = {
        "logfilepath": "video.log",
        "proxies": {},
        "savedir": "download"
    }
    
    video_client = videoall.videodl(config=config)
    
    video_path = 'the video path'
    
    video_client.get_video(video_path)
